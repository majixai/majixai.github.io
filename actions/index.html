Quick Reference

// 1. Configure (before script load)
window.ACTIONS_CONFIG = {
  namespace:    'myapp',
  logToConsole: true,
  persistHistory: false,
};

// 2. Initialize
MajixActions.init();

// 3. Register handlers
var off = MajixActions.on('data/fetch', function (payload) {
  return fetch('/api?ticker=' + payload.ticker).then(r => r.json());
});
MajixActions.once('app/ready', function () { console.log('ready once'); });

// 4. Dispatch
MajixActions.dispatch('data/fetch', { ticker: 'AAPL' });
await MajixActions.dispatchAsync('data/save', { rows: [...] });

// 5. Middleware
MajixActions.use(function (action, next) {
  console.time(action.type);
  next(action);
  console.timeEnd(action.type);
});

// 6. Inspect
MajixActions.history(10);     // last 10 actions (most recent first)
MajixActions.hasHandler('*'); // true / false
off();                        // unsubscribe
