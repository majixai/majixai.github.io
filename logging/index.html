Quick Reference

// 1. Configure (before script load)
window.LOGGER_CONFIG = {
  appName: 'myapp',
  level:   'info',
  targets: ['console','memory','localStorage'],
};

// 2. Initialize
MajixLogger.init();

// 3. Write entries
MajixLogger.debug('verbose detail', { val: 42 });
MajixLogger.info('User logged in', { userId: 1 }, 'auth');
MajixLogger.warn('Slow response', { ms: 2400 }, 'network');
MajixLogger.error('Fetch failed', { status: 500 });

// 4. Performance timers
var stop = MajixLogger.time('data-fetch');
fetch('/api/data').then(stop);

// 5. Inspect / export
MajixLogger.getEntries({ level: 'warn' });
MajixLogger.exportJSON();
MajixLogger.queryDB({ level: 'error', limit: 50 })
           .then(console.table);

// 6. Flush remote queue
window.addEventListener('beforeunload', MajixLogger.flush);
