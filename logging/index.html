Overview

logging-core.js is a global, IIFE-style library (no ES modules) that provides structured logging for every MajixAI page and script. Include the file once, set window.LOGGER_CONFIG, call MajixLogger.init(), and all five output targets are immediately available.

Targets: console memory localStorage indexedDB remote

Quick start

<!-- 1. Include the library -->
<script src="/logging/logging-core.js"></script>

<!-- 2. Configure and initialise -->
<script>
window.LOGGER_CONFIG = {
  namespace: 'my-app',
  level:     'DEBUG',
  targets:   ['console', 'memory'],
};
MajixLogger.init();
</script>
// 3. Log messages
MajixLogger.info('Page loaded');
MajixLogger.warn('Slow network', { latencyMs: 1200 });
MajixLogger.error('Fetch failed', { url: '/api/data', status: 500 });

// Child logger with inherited namespace + metadata
const auth = MajixLogger.child('auth', { userId: 42 });
auth.info('Login successful');

// Perf timer
MajixLogger.time('fetchData');
// … async work …
MajixLogger.timeEnd('fetchData');
// → DEBUG [app] fetchData: 123.456ms

Output targets

TargetDescription
'console'Colour-coded output to window.console, using console.debug/log/warn/error by level.
'memory'In-process ring buffer bounded by bufferSize. Query with MajixLogger.query(filter).
'localStorage'JSON array persisted to localStorage[localStorageKey]. Trimmed to localStorageMax entries.
'indexedDB'Persisted to IndexedDB with level/namespace/ts indexes. Query with MajixLogger.queryDB(filter, cb).
'remote'Batched HTTP POST (JSON) to remoteUrl. Auto-flushes at remoteBatchSize or remoteFlushMs.

Log levels

LevelNumericUse case
TRACE0Very verbose diagnostic information.
DEBUG1Detailed debug information.
INFO2Normal operational messages.
WARN3Recoverable issues worth monitoring.
ERROR4Unexpected errors that may impact functionality.
FATAL5Critical failures requiring immediate attention.

LOGGER_CONFIG reference

levelstring'DEBUG'Minimum log level.
targetsstring[]['console','memory']Active output targets.
bufferSizenumber500Max entries in the memory ring buffer.
remoteUrlstring''HTTP POST endpoint for the remote target.
remoteBatchSizenumber10Entries before auto-flush to remote.
remoteFlushMsnumber5000Max ms between remote flushes.
localStorageKeystring'majix_log'localStorage key name.
localStorageMaxnumber200Max entries kept in localStorage.

includeStackbooleanfalseAttach stack trace to WARN/ERROR/FATAL.
metaobject{}Static fields merged into every entry.

API reference

Logging

MajixLogger.trace(msg, data?)
MajixLogger.debug(msg, data?)
MajixLogger.info(msg, data?)
MajixLogger.warn(msg, data?)
MajixLogger.error(msg, data?)
MajixLogger.fatal(msg, data?)
MajixLogger.log(level, msg, data?)   // generic — level as string

Child loggers

const db = MajixLogger.child('db', { poolId: 1 });
db.info('Connection acquired');
// → namespace: 'app:db', poolId: 1

const query = db.child('query');
query.debug('Executing', { sql: 'SELECT ...' });
// → namespace: 'app:db:query'

Performance timers

MajixLogger.time('init');
// … work …
MajixLogger.timeEnd('init');
// → DEBUG [app] init: 42.100ms  { elapsed_ms: 42.1 }

Memory buffer

// All entries
MajixLogger.query();

// Filtered
MajixLogger.query({ level: 'WARN' });
MajixLogger.query({ namespace: 'auth', limit: 10 });
MajixLogger.query({ search: 'failed', since: Date.now() - 60000 });

// JSON export
const json = MajixLogger.exportJSON();

IndexedDB

MajixLogger.queryDB({ level: 'ERROR', limit: 50 }, function (err, entries) {
  console.table(entries);
});

MajixLogger.clearDB(function (err) {
  console.log('DB cleared');
});

Remote

MajixLogger.flush();   // immediately POST pending batch

Runtime configuration

MajixLogger.setLevel('WARN');
MajixLogger.addTarget('localStorage');
MajixLogger.removeTarget('console');

Event bus

MajixLogger.on('entry', function (entry) {
  // called synchronously for every dispatched entry
});

MajixLogger.on('flush', function ({ count, url }) {
  // called after each remote POST
});

MajixLogger.on('error', function (err) {
  // internal errors (DB failures, fetch errors, …)
});

MajixLogger.off('entry', handler);

Introspection

MajixLogger.getConfig();   // copy of current config
MajixLogger.getBuffer();   // copy of memory buffer array
MajixLogger.getLevels();   // ['TRACE','DEBUG','INFO','WARN','ERROR','FATAL']

Usage examples

Basic app logging

window.LOGGER_CONFIG = {
  namespace: 'dashboard',
  level:     'INFO',
  targets:   ['console', 'memory'],
  meta:      { appVersion: '2.1.0' },
};
MajixLogger.init();

MajixLogger.info('Dashboard ready');

Persist logs across reloads

window.LOGGER_CONFIG = {
  namespace: 'tracker',
  targets:   ['console', 'memory', 'localStorage', 'indexedDB'],
  level:     'DEBUG',
};
MajixLogger.init();

Send errors to a remote endpoint

window.LOGGER_CONFIG = {
  namespace:       'trading',
  level:           'INFO',
  targets:         ['console', 'memory', 'remote'],
  remoteUrl:       'https://logs.example.com/ingest',
  remoteBatchSize: 5,
  remoteFlushMs:   10000,
};
MajixLogger.init();

Structured logging with child loggers

MajixLogger.init({ namespace: 'api', level: 'DEBUG', targets: ['console', 'memory'] });

const req = MajixLogger.child('request', { requestId: 'abc-123' });
req.info('Received');
req.debug('Validating payload', { bodySize: 512 });
req.warn('Rate limit approaching', { remaining: 3 });

Stack trace capture

window.LOGGER_CONFIG = {
  namespace:    'app',
  includeStack: true,
  targets:      ['console', 'memory'],
};
MajixLogger.init();

MajixLogger.error('Unexpected state', { code: 'E_INVALID' });
