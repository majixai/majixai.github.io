//@version=5
// =============================================================================
// JINXAI Tensor Forecast  |  Pine Script v5
// T ∈ R^(n × m × p) — Time × Features × Regime States
//
// Architecture:
//   Layer 1 — Feature Matrix  F ∈ R^(n × m)
//   Layer 2 — Tensor Core  C = FᵀF / n  ∈ R^(m × m)  (covariance)
//   Layer 3 — HOSVD Smoothing via truncated eigendecomposition
//   Layer 4 — Regime Classification  P ∈ R^(n × p)  via softmax
//   Layer 5 — Forecast Path with σ-bands (Gaussian Probability Field)
// =============================================================================
indicator("JINXAI Tensor Forecast", overlay=true,
          max_labels_count=500, max_lines_count=200, max_boxes_count=50)

// ─────────────────────────────────────────────────────────────────────────────
// INPUTS
// ─────────────────────────────────────────────────────────────────────────────
var GRP_T = "Tensor Dimensions  T ∈ R^(n × m × p)"
int    N  = input.int(20,  "n — Lookback  (Time Axis)",    minval=5,  maxval=100, group=GRP_T)
int    H  = input.int(10,  "Horizon (forecast bars ahead)",minval=1,  maxval=30,  group=GRP_T)

var GRP_W = "Feature Weights  w₁…w₄  (Price · RSI · Volatility · Momentum)"
float  w1 = input.float(0.40, "w₁  Price",      minval=0.0, maxval=1.0, step=0.05, group=GRP_W)
float  w2 = input.float(0.25, "w₂  RSI",        minval=0.0, maxval=1.0, step=0.05, group=GRP_W)
float  w3 = input.float(0.20, "w₃  Volatility", minval=0.0, maxval=1.0, step=0.05, group=GRP_W)
float  w4 = input.float(0.15, "w₄  Momentum",   minval=0.0, maxval=1.0, step=0.05, group=GRP_W)

var GRP_V = "Visualisation"
bool   showForecast = input.bool(true,    "Show Tensor Forecast Line",  group=GRP_V)
bool   showBands    = input.bool(true,    "Show Probability Field (σ)",  group=GRP_V)
bool   showLabels   = input.bool(true,    "Show Horizon Label",          group=GRP_V)
bool   showRegime   = input.bool(true,    "Show Regime Dot Markers",     group=GRP_V)
bool   showTable    = input.bool(true,    "Show Summary Table",          group=GRP_V)
color  cForecast    = input.color(#00ff88,"Forecast Line",               group=GRP_V)
color  cField       = input.color(#00ff88,"Probability Field Tint",      group=GRP_V)
float  sigmaScale   = input.float(1.0,    "σ Scale",    minval=0.1, maxval=3.0, step=0.1, group=GRP_V)

// ─────────────────────────────────────────────────────────────────────────────
// SECTION 1 — FEATURE MATRIX  F ∈ R^(n × m)
//
// Each row i is the feature vector for bar[i] (0 = current, N-1 = oldest).
// m = 4 features:
//   col 0  Price z-score:       (close[i] - μ) / σ
//   col 1  Normalised RSI:      (RSI[i] - 50) / 50
//   col 2  ATR normalised:      atr[i] / atr  (ratio to current ATR)
//   col 3  Price momentum:      (close[i] - close[i+N]) / close[i+N]
// ─────────────────────────────────────────────────────────────────────────────

// Running μ and σ over the lookback window
float mu_price  = ta.sma(close,   N)
float sd_price  = ta.stdev(close, N) + 1e-9
float atr_now   = ta.atr(14)

// Feature column vectors (length N arrays)
var float[] fPrice = array.new_float(0)
var float[] fRsi   = array.new_float(0)
var float[] fAtr   = array.new_float(0)
var float[] fMom   = array.new_float(0)

array.clear(fPrice)
array.clear(fRsi)
array.clear(fAtr)
array.clear(fMom)

for i = 0 to N - 1
    // col 0: z-score price
    array.push(fPrice, (close[i] - mu_price) / sd_price)
    // col 1: centred & scaled RSI
    array.push(fRsi,   (ta.rsi(close, 14)[i] - 50.0) / 50.0)
    // col 2: ATR ratio (volatility regime)
    float atr_i = ta.atr(14)[i]
    array.push(fAtr,   atr_now > 0 ? (atr_i / atr_now) - 1.0 : 0.0)
    // col 3: N-bar momentum
    float base = close[i + N]
    array.push(fMom,   base != 0 ? (close[i] - base) / base : 0.0)

// ─────────────────────────────────────────────────────────────────────────────
// SECTION 2 — TENSOR CORE  C = FᵀF / N  ∈ R^(4 × 4)  (covariance matrix)
//
// We compute only the dot-products needed for the weight-projected score.
// Full covariance row 0 (price feature): [c00, c01, c02, c03]
// ─────────────────────────────────────────────────────────────────────────────
f_dot(a, b) =>
    float s = 0.0
    int   sz = math.min(array.size(a), array.size(b))
    for j = 0 to sz - 1
        s += array.get(a, j) * array.get(b, j)
    s / sz

// Covariance matrix row 0 (price row) — used for the dominant-eigenvector projection
float c00 = f_dot(fPrice, fPrice)   // price · price
float c01 = f_dot(fPrice, fRsi)     // price · RSI
float c02 = f_dot(fPrice, fAtr)     // price · volatility
float c03 = f_dot(fPrice, fMom)     // price · momentum

// Cross-feature covariance entries for HOSVD noise estimate
float c11 = f_dot(fRsi, fRsi)
float c22 = f_dot(fAtr, fAtr)
float c33 = f_dot(fMom, fMom)

// ─────────────────────────────────────────────────────────────────────────────
// SECTION 3 — HOSVD NOISE FLOOR + DENOISED DRIFT
//
// Standard HOSVD truncates small singular values.  Here we approximate the
// noise floor as the median off-diagonal covariance magnitude and use it to
// compute a soft shrinkage coefficient α ∈ [0, 1].
//
// The denoised drift  Δ_smooth = α · Δ_raw + (1 − α) · SMA(Δ_raw, N)
// ─────────────────────────────────────────────────────────────────────────────
float noiseFloor    = (math.abs(c01) + math.abs(c02) + math.abs(c03)) / 3.0
float diagTrace     = c00 + c11 + c22 + c33 + 1e-9          // proxy for total variance
float alpha         = math.max(0.0, 1.0 - noiseFloor / diagTrace)  // shrinkage

float rawDrift      = (close - close[N]) / N
float sma_drift     = ta.sma(rawDrift, N)
float smoothDrift   = alpha * rawDrift + (1.0 - alpha) * sma_drift

// ─────────────────────────────────────────────────────────────────────────────
// SECTION 4 — TENSOR CONTRACTION  →  scalar score
//
// score = w₁·c00 + w₂·c01 + w₃·c02 + w₄·c03
//       = feature-weight vector projected onto the price covariance row.
// Equivalent to one power-method iteration for the leading eigenvector.
// ─────────────────────────────────────────────────────────────────────────────
float tensorScore  = w1 * c00 + w2 * c01 + w3 * c02 + w4 * c03
float sigmoidScore = 1.0 / (1.0 + math.exp(-tensorScore * 5.0))   // ∈ (0, 1)

// ─────────────────────────────────────────────────────────────────────────────
// SECTION 5 — REGIME CLASSIFICATION  P ∈ R^(n × p)
//
// Three-class soft labels via sigmoid:
//   p_bull   = σ(score · 5)
//   p_bear   = 1 − p_bull
//   p_neutral = 1 − |p_bull − 0.5| · 2     (high when near 0.5)
// ─────────────────────────────────────────────────────────────────────────────
float pBull    = sigmoidScore
float pBear    = 1.0 - sigmoidScore
float pNeutral = 1.0 - math.abs(sigmoidScore - 0.5) * 2.0

// Dominant regime label
string regime = pBull > pBear and pBull > pNeutral ? "BULL" :
                pBear > pBull and pBear > pNeutral ? "BEAR" : "NEUT"
color  rColor = regime == "BULL" ? color.new(color.green, 30) :
                regime == "BEAR" ? color.new(color.red,   30) :
                                   color.new(color.gray,  50)

// ─────────────────────────────────────────────────────────────────────────────
// SECTION 6 — FORECAST PATH
//
// Tensor-adjusted drift:  Δ_adj = Δ_smooth × (1 + ½·tanh(score))
// Centroid at horizon k:  E[p_{t+k}] = p_t + Δ_adj · k
// Uncertainty band:       σ_k = ATR × √k × sigmaScale
// ─────────────────────────────────────────────────────────────────────────────
float adjFactor   = 1.0 + 0.5 * math.tanh(tensorScore)
float adjustedDrift = smoothDrift * adjFactor

float forecastH   = close + adjustedDrift * H
float upperH      = forecastH + atr_now * math.sqrt(H) * sigmaScale
float lowerH      = forecastH - atr_now * math.sqrt(H) * sigmaScale

// ─────────────────────────────────────────────────────────────────────────────
// SECTION 7 — VISUALISATION
// ─────────────────────────────────────────────────────────────────────────────

// ── Tensor Forecast centroid line ──
plot(showForecast ? forecastH : na, "Tensor Forecast (centroid)",
     color=cForecast, linewidth=2, style=plot.style_linebr)

// ── Gaussian Probability Field  (σ corridor) ──
p_upper = plot(showBands ? upperH : na, "σ Upper Band",
               color=color.new(cField, 60), linewidth=1, style=plot.style_linebr)
p_lower = plot(showBands ? lowerH : na, "σ Lower Band",
               color=color.new(cField, 60), linewidth=1, style=plot.style_linebr)
fill(p_upper, p_lower, color=color.new(cField, 85), title="Tensor Probability Field")

// ── Denoised price signal (lookback reconstructed) ──
float denoisedClose = close + smoothDrift
plot(denoisedClose, "Denoised Price (HOSVD)",
     color=color.new(#8888ff, 60), linewidth=1, style=plot.style_line)

// ── Regime dot markers ──
if showRegime
    plotshape(true, "Regime Marker", shape.circle,
              location.abovebar, color=rColor, size=size.tiny)

// ── Horizon forecast label ──
if barstate.islast and showLabels
    color lbColor = regime == "BULL" ? color.new(#00ff88, 5) :
                    regime == "BEAR" ? color.new(#ff4444, 5) :
                                       color.new(#aaaaaa, 5)
    label.new(
        bar_index + H, forecastH,
        "⊕ " + str.tostring(math.round(forecastH, 2)) + "\n" +
        "σ+ " + str.tostring(math.round(upperH - forecastH, 2)) + "\n" +
        "σ− " + str.tostring(math.round(forecastH - lowerH, 2)) + "\n" +
        "Regime: " + regime,
        color=lbColor, textcolor=color.black,
        style=label.style_label_left, size=size.small)

    // Draw sigma lines to horizon
    line.new(bar_index, close, bar_index + H, forecastH,
             color=color.new(cForecast, 30), width=1, style=line.style_dashed)
    line.new(bar_index, close, bar_index + H, upperH,
             color=color.new(cField, 50), width=1, style=line.style_dotted)
    line.new(bar_index, close, bar_index + H, lowerH,
             color=color.new(cField, 50), width=1, style=line.style_dotted)

// ── Summary Table ──
if showTable and barstate.islast
    var table tbl = table.new(position.top_right, 2, 9,
                              border_width=1,
                              border_color=color.new(color.gray, 50),
                              frame_color=color.new(color.gray, 20),
                              frame_width=1)

    // Header
    table.cell(tbl, 0, 0, "JINXAI Tensor",
               bgcolor=color.new(#00ff88, 15), text_color=color.black, text_size=size.small)
    table.cell(tbl, 1, 0, "Value",
               bgcolor=color.new(#00ff88, 15), text_color=color.black, text_size=size.small)

    // Tensor Score
    table.cell(tbl, 0, 1, "T-Score",      text_color=color.white, text_size=size.small)
    table.cell(tbl, 1, 1, str.tostring(math.round(tensorScore, 4)),
               text_color=color.white, text_size=size.small)

    // HOSVD alpha
    table.cell(tbl, 0, 2, "HOSVD α",      text_color=color.white, text_size=size.small)
    table.cell(tbl, 1, 2, str.tostring(math.round(alpha, 4)),
               text_color=color.white, text_size=size.small)

    // Regime
    table.cell(tbl, 0, 3, "Regime",        text_color=color.white, text_size=size.small)
    table.cell(tbl, 1, 3, regime, bgcolor=rColor, text_color=color.white, text_size=size.small)

    // Regime probabilities
    table.cell(tbl, 0, 4, "p(Bull)",       text_color=color.white, text_size=size.small)
    table.cell(tbl, 1, 4, str.tostring(math.round(pBull * 100, 1)) + "%",
               text_color=color.new(color.green, 10), text_size=size.small)

    table.cell(tbl, 0, 5, "p(Bear)",       text_color=color.white, text_size=size.small)
    table.cell(tbl, 1, 5, str.tostring(math.round(pBear * 100, 1)) + "%",
               text_color=color.new(color.red, 10), text_size=size.small)

    table.cell(tbl, 0, 6, "Adj Factor",    text_color=color.white, text_size=size.small)
    table.cell(tbl, 1, 6, str.tostring(math.round(adjFactor, 4)),
               text_color=color.white, text_size=size.small)

    // Forecast at horizon H
    table.cell(tbl, 0, 7, "Forecast +" + str.tostring(H) + "b",
               text_color=color.white, text_size=size.small)
    table.cell(tbl, 1, 7, str.tostring(math.round(forecastH, 4)),
               text_color=color.new(#00ff88, 0), text_size=size.small)

    // ATR / σ
    table.cell(tbl, 0, 8, "σ (ATR×√H)",   text_color=color.white, text_size=size.small)
    table.cell(tbl, 1, 8,
               str.tostring(math.round(atr_now * math.sqrt(H) * sigmaScale, 4)),
               text_color=color.white, text_size=size.small)

// ─────────────────────────────────────────────────────────────────────────────
// ALERTS
// ─────────────────────────────────────────────────────────────────────────────
alertcondition(regime == "BULL" and regime[1] != "BULL",
               title="Tensor → BULL Regime",
               message="{{ticker}} Tensor regime shifted to BULLISH. Score={{plot_0}}")
alertcondition(regime == "BEAR" and regime[1] != "BEAR",
               title="Tensor → BEAR Regime",

alertcondition(close > upperH,

macd_fast_length = input(12, title="MACD Fast Length")

rsi_value = ta.rsi(close, rsi_period)

// MACD Calculation
[macd_line, signal_line, _] = ta.macd(close, macd_fast_length, macd_slow_length, macd_signal_length)

// Bollinger Bands Calculation
[bb_upper, bb_middle, bb_lower] = ta.bb(close, bb_length, bb_std_dev)

// Volume Analysis
volume_signal = volume > volume_threshold ? 1 : 0

// Neural Network Scoring System (dummy implementation)
score = (rsi_value / 100) + (macd_line - signal_line) + (close - bb_middle) / (bb_upper - bb_lower) + volume_signal

// Buy/Sell Thresholds
buy_threshold = input(2.0, title="Buy Threshold")
sell_threshold = input(-2.0, title="Sell Threshold")

// Buy/Sell Signals
buy_signal = score >= buy_threshold
sell_signal = score <= sell_threshold

// Plotting Signals
plotshape(buy_signal, style=shape.labelup, location=location.belowbar, color=color.green, size=size.small, title="Buy Signal")
plotshape(sell_signal, style=shape.labeldown, location=location.abovebar, color=color.red, size=size.small, title="Sell Signal")

smDMPlus  = ta.rma(dmPlus,    adxLen)
smDMMinus = ta.rma(dmMinus,   adxLen)

diPlus    = smTR != 0 ? 100 * smDMPlus  / smTR : 0
diMinus   = smTR != 0 ? 100 * smDMMinus / smTR : 0
dx        = (diPlus + diMinus) != 0 ? 100 * math.abs(diPlus - diMinus) / (diPlus + diMinus) : 0
adxVal    = ta.rma(dx, adxLen)

trendStrong = adxVal > adxThresh
bullTrend   = diPlus  > diMinus and trendStrong
bearTrend   = diMinus > diPlus  and trendStrong

// VWAP standard deviation bands
vwapDev   = ta.stdev(hlc3, 20)
vwapHigh  = vwapVal + vwapBandMult * vwapDev
vwapLow   = vwapVal - vwapBandMult * vwapDev
aboveVWAP = close > vwapVal
belowVWAP = close < vwapVal

ema21     = ta.ema(close, 21)
ema50     = ta.ema(close, 50)
ema200    = ta.ema(close, 200)
emaRibbonBull = ema8 > ema21 and ema21 > ema50
emaRibbonBear = ema8 < ema21 and ema21 < ema50

highVol   = volume > volSma * 1.5
lowVol    = volume < volSma * 0.5

obvBear   = obv < obvEma

// SECTION 3 · CANDLESTICK PATTERN DETECTION

// Helper values
bodySize   = math.abs(close - open)
fullRange  = high - low
upperWick  = close >= open ? high - close : high - open
lowerWick  = close >= open ? open - low   : close - low
isBull     = close > open
isBear     = close < open

// Doji
isDoji = bodySize / fullRange < dojiThresh and fullRange > 0

// Hammer  (small body near top, long lower wick, tiny upper wick)
isHammer = lowerWick >= 2.0 * bodySize and upperWick <= 0.3 * bodySize and
           bodySize > 0 and not isBull[1]

// Inverted Hammer
isInvHammer = upperWick >= 2.0 * bodySize and lowerWick <= 0.3 * bodySize and bodySize > 0

// Shooting Star
isShootingStar = upperWick >= 2.0 * bodySize and lowerWick <= 0.3 * bodySize and
                 isBear and (high[1] < high)

// Hanging Man
isHangingMan = lowerWick >= 2.0 * bodySize and upperWick <= 0.3 * bodySize and
               isBear and (close[1] > close)

// Bullish Engulfing
isBullEngulfing = isBull and isBear[1] and
                  open <= close[1] and close >= open[1]

// Bearish Engulfing
isBearEngulfing = isBear and isBull[1] and
                  open >= close[1] and close <= open[1]

// Morning Star (3-bar)
isMorningStar = isBear[2] and isDoji[1] and isBull and
                close > (open[2] + close[2]) / 2

// Evening Star (3-bar)
isEveningStar = isBull[2] and isDoji[1] and isBear and
                close < (open[2] + close[2]) / 2

// Three White Soldiers
is3WhiteSoldiers = isBull and isBull[1] and isBull[2] and
                   close > close[1] and close[1] > close[2] and
                   open > open[1] and open[1] > open[2]

// Three Black Crows
is3BlackCrows = isBear and isBear[1] and isBear[2] and
                close < close[1] and close[1] < close[2] and
                open < open[1] and open[1] < open[2]

// Bullish Harami
isBullHarami = isBull and isBear[1] and
               open > close[1] and close < open[1]

// Bearish Harami
isBearHarami = isBear and isBull[1] and
               open < close[1] and close > open[1]

// Dark Cloud Cover
isDarkCloud = isBear and isBull[1] and
              open > high[1] and close < (open[1] + close[1]) / 2

// Piercing Line
isPiercing = isBull and isBear[1] and
             open < low[1] and close > (open[1] + close[1]) / 2

// Bullish pattern count
bullPatterns = (isHammer        ? 1 : 0) +
               (isBullEngulfing ? 1 : 0) +
               (isMorningStar   ? 1 : 0) +
               (is3WhiteSoldiers? 1 : 0) +
               (isBullHarami    ? 1 : 0) +
               (isPiercing      ? 1 : 0) +
               (isInvHammer     ? 1 : 0)

// Bearish pattern count
bearPatterns = (isShootingStar  ? 1 : 0) +
               (isHangingMan    ? 1 : 0) +
               (isBearEngulfing ? 1 : 0) +

// When useSeedData is enabled and a valid URL is provided, this block fetches
// an external CSV / .dat file published via Pine Seeds and extracts the most
// recent price entry to overlay on the chart.

var float seedPrice = na

if useSeedData and str.length(seedUrl) > 0
    var string rawSeed = request.seed(seedUrl, "")
    if not na(rawSeed) and str.length(rawSeed) > 0
        var string[] seedLines = str.split(rawSeed, "\n")
        int nLines = array.size(seedLines)
        if nLines > 1
            // Walk backwards to find the last non-empty data line
            int idx = nLines - 1
            while idx > 0
                string ln = array.get(seedLines, idx)
                if str.length(ln) > 0
                    string[] parts = str.split(ln, ",")
                    if array.size(parts) >= 2
                        float p = str.tonumber(array.get(parts, 1))
                        if not na(p)
                            seedPrice := p
                            break
                idx := idx - 1

// Plot external seed price as a dashed horizontal reference

     color=color.new(color.purple, 0), style=plot.style_circles,
     linewidth=2)

// ---- Skew-StdDev Bias Engine -----------------------------------------------
// Rolling return statistics used to adaptively bias all pattern scores.
// Positive skewness → price has fat right tail → favour bullish continuations.
// Negative skewness → fat left tail            → favour bearish reversals.
// High short-term vol vs long-term vol         → amplify reversal scores.
// Low  short-term vol vs long-term vol         → amplify continuation scores.
int   _skewLen    = 20
float _ret        = close / nz(close[1], close) - 1.0
float _retMean    = ta.sma(_ret, _skewLen)
float _retStd     = ta.stdev(_ret, _skewLen)
float _retVar     = _retStd * _retStd
float _cubedDev   = ta.sma(math.pow(_ret - _retMean, 3), _skewLen)
float skewness    = _retStd > 1e-10 ? _cubedDev / math.pow(_retStd, 3) : 0.0
float _volShort   = ta.stdev(_ret, _skewLen)
float _volLong    = ta.stdev(_ret, _skewLen * 3)
float stdevMult   = math.max(0.7, math.min(1.5, _volLong > 1e-10 ? 1.0 + (1.0 - _volShort / _volLong) * 0.5 : 1.0))
int   skewBias    = int(math.round(math.max(-3.0, math.min(3.0, skewness * 2.0))))

// Kurtosis (excess) for fat-tail regime detection
float _kurtRaw    = ta.sma(math.pow(_ret - _retMean, 4), _skewLen)
float kurtosis    = _retVar > 1e-10 ? _kurtRaw / (_retVar * _retVar) - 3.0 : 0.0

plb(int idx) => array.size(pivotsLow)  > idx ? array.get(pivotsLow,  idx).barIndex : na

dist(float a, float b) =>
    absMagnitude = math.max(math.abs(a), 1e-10)
    math.abs(a - b) / absMagnitude

fibRatio(float move1, float move2) =>
    move1 == 0 ? 0.0 : math.abs(move2 / move1)

drawGeometricLines(int x1, float y1, int x2, float y2, int x3, float y3, color c) =>

        line.new(x2, y2, x3, y3, color=c, width=2)

    else if dist(pl(1), pl(0)) < st_tl and ph(0) > pl(0) + ((ph(1) - pl(1)) * 0.6)

        else if pl(1) > pl(0) and ph(1) > ph(0) and dist(plb(1), phb(0)) > 8

    if mScore == 0
        if (ph(1) - pl(1)) / math.max(pl(1), 1e-10) > 0.04 and dist(ph(1), ph(0)) < st_tl and math.abs(phb(1) - phb(0)) <= 5

        else if (ph(1) - pl(1)) / math.max(pl(1), 1e-10) > 0.04 and dist(pl(1), pl(0)) < st_tl and math.abs(plb(1) - plb(0)) <= 5

    bool flatTops       = dist(ph(0), ph(1)) < st_tl
    bool flatBottoms    = dist(pl(0), pl(1)) < st_tl
    bool risingTops     = ph(0) > ph(1) and not flatTops
    bool fallingTops    = ph(0) < ph(1) and not flatTops
    bool risingBottoms  = pl(0) > pl(1) and not flatBottoms
    bool fallingBottoms = pl(0) < pl(1) and not flatBottoms

    float topDiff   = math.abs(ph(0) - ph(1))
    float botDiff   = math.abs(pl(0) - pl(1))
    bool isParallel = math.abs(topDiff - botDiff) < (math.max(topDiff, botDiff) * 0.3)

    // 51 & 52. Double Top / Bottom
    if flatTops and not flatBottoms
        mScore := -9, mName := "Double Top", mTarget := pl(0) - (ph(0) - pl(0))
    else if flatBottoms and not flatTops
        mScore := 9, mName := "Double Bottom", mTarget := ph(0) + (ph(0) - pl(0))

    // 53 & 54. Rectangle / Box Play
    else if flatTops and flatBottoms
        mScore := 0, mName := "Rectangle / Ranging", mTarget := ph(0)

    // 55 & 56. Ascending / Descending Triangle
    else if flatTops and risingBottoms
        mScore := 8, mName := "Ascending Triangle", mTarget := ph(0) + (ph(0) - pl(1))
    else if flatBottoms and fallingTops
        mScore := -8, mName := "Descending Triangle", mTarget := pl(0) - (ph(1) - pl(0))

    // 57 & 58. Symmetrical Triangle
    else if fallingTops and risingBottoms
        mScore := (close > ta.ema(close, 50)) ? 5 : -5, mName := "Symmetrical Triangle", mTarget := ph(0)

    // 59 & 60. Broadening Megaphone
    else if risingTops and fallingBottoms
        mScore := 0, mName := "Broadening Megaphone", mTarget := na

    // 61 & 62. Right-Angled Broadening
    else if flatBottoms and risingTops
        mScore := 6, mName := "Rt-Angle Asc Broadening", mTarget := ph(0) + (ph(0) - pl(0))
    else if flatTops and fallingBottoms
        mScore := -6, mName := "Rt-Angle Desc Broadening", mTarget := pl(0) - (ph(0) - pl(0))

    // 63 & 64. Wedges
    else if risingTops and risingBottoms and topDiff < botDiff and not isParallel
        mScore := -8, mName := "Rising Wedge", mTarget := pl(1)
    else if fallingTops and fallingBottoms and topDiff > botDiff and not isParallel
        mScore := 8, mName := "Falling Wedge", mTarget := ph(1)

    // 65 & 66. Flags
    else if fallingTops and fallingBottoms and isParallel and math.abs(phb(1) - plb(0)) < 15
        mScore := 7, mName := "Bull Flag", mTarget := ph(1) + (ph(1) - pl(1))
    else if risingTops and risingBottoms and isParallel and math.abs(plb(1) - phb(0)) < 15
        mScore := -7, mName := "Bear Flag", mTarget := pl(1) - (ph(1) - pl(1))

    // 67 & 68. Channels
    else if fallingTops and fallingBottoms and isParallel
        mScore := 6, mName := "Channel Down", mTarget := ph(1)
    else if risingTops and risingBottoms and isParallel
        mScore := -6, mName := "Channel Up", mTarget := pl(1)

    // 69 & 70. AB=CD Harmonic
    else if fallingTops and fallingBottoms and dist(topDiff, botDiff) < 0.15
        mScore := 10, mName := "AB=CD (Bullish)", mTarget := ph(0)
    else if risingTops and risingBottoms and dist(topDiff, botDiff) < 0.15
        mScore := -10, mName := "AB=CD (Bearish)", mTarget := pl(0)

    else if fallingTops and risingBottoms and (ph(1) - pl(1)) / math.max(pl(1), 1e-10) > 0.05
        mScore := 10, mName := "Bullish Pennant", mTarget := ph(0) + (ph(1) - pl(1))
    else if fallingTops and risingBottoms and array.size(pivotsHigh) >= 3 and (pl(1) - ph(2)) / math.max(ph(2), 1e-10) < -0.05

    else if flatTops and math.abs(phb(1) - phb(0)) <= 3

    else if flatBottoms and math.abs(plb(1) - plb(0)) <= 3
        mScore := 11, mName := "Horn Bottom", mTarget := ph(0)

    // 75 & 76. Pipe Tops & Bottoms (Extended sharp double pivots)
    else if flatTops and math.abs(phb(1) - phb(0)) >= 4 and math.abs(phb(1) - phb(0)) <= 8 and (ph(0) - pl(0)) / math.max(pl(0), 1e-10) > 0.05
        mScore := -12, mName := "Pipe Top", mTarget := pl(0) - (ph(0) - pl(0))

        mScore := 12, mName := "Pipe Bottom", mTarget := ph(0) + (ph(0) - pl(0))

    // 77 & 78. Darvas Box / Squeeze (very tight rectangle < 2% height)
    else if flatTops and flatBottoms and (ph(0) - pl(0)) / math.max(pl(0), 1e-10) < 0.02
        mScore := 5, mName := "Darvas Box / Squeeze", mTarget := ph(0) + (ph(0) - pl(0))

    // 79 & 80. Hook Reversal
    else if ph(0) < ph(1) and pl(0) > pl(1) and close < open and close[1] > open[1]
        mScore := -8, mName := "Hook Reversal Top", mTarget := pl(1)
    else if ph(0) < ph(1) and pl(0) > pl(1) and close > open and close[1] < open[1]
        mScore := 8, mName := "Hook Reversal Bottom", mTarget := ph(1)

// ============================================================================
// LEVEL 5: ADVANCED HARMONICS — Patterns 81-100
//          Shark, Cypher, 5-0, Alt-Bat, Ideal-Bat, Nenstar, BAMM, Leonardo
// ============================================================================

if mScore == 0 and array.size(pivotsHigh) >= 3 and array.size(pivotsLow) >= 3

    // 81 & 82. Shark Harmonic (0-X-A-B-C structure; D at 0.886-1.13 of OX)
    float _OX_s   = math.abs(ph(2) - pl(2))
    float _XC_s_b = math.abs(pl(0) - ph(2))
    float _XC_s_r = math.abs(ph(0) - pl(2))
    float _shRet_b = fibRatio(_OX_s, _XC_s_b)
    float _shRet_r = fibRatio(_OX_s, _XC_s_r)
    if ph(2) > pl(2) and pl(1) < ph(2) and ph(1) > pl(1) and pl(0) < ph(1)
        if _shRet_b >= 0.886 and _shRet_b <= 1.13
            mScore := 22, mName := "Bullish Shark", mTarget := ph(1) * 1.13
    if mScore == 0 and pl(2) < ph(2) and ph(1) > pl(2) and pl(1) < ph(1) and ph(0) > pl(1)
        if _shRet_r >= 0.886 and _shRet_r <= 1.13
            mScore := -22, mName := "Bearish Shark", mTarget := pl(1) * 0.886

    // 83 & 84. Cypher Pattern (AB = 0.382-0.618 of XA; D = 0.786 retracement of XC)
    if mScore == 0
        float _XA_cy   = ph(1) - pl(2)
        float _AB_cy   = math.abs(pl(1) - ph(1))
        float _BC_cy   = ph(0) - pl(1)
        float _abRet   = fibRatio(_XA_cy, _AB_cy)
        float _bcExt   = fibRatio(_XA_cy, _BC_cy)
        if _abRet >= 0.382 and _abRet <= 0.618 and _bcExt >= 1.272 and _bcExt <= 1.414
            if pl(0) < pl(1)
                mScore := 20, mName := "Bullish Cypher", mTarget := ph(0) * 1.786 - pl(0) * 0.786

        float _XA_cy2  = pl(1) - ph(2)
        float _AB_cy2  = math.abs(ph(1) - pl(1))
        float _BC_cy2  = math.abs(pl(0) - ph(1))
        float _abRet2  = fibRatio(math.abs(_XA_cy2), _AB_cy2)
        float _bcExt2  = fibRatio(math.abs(_XA_cy2), _BC_cy2)
        if mScore == 0 and _abRet2 >= 0.382 and _abRet2 <= 0.618 and _bcExt2 >= 1.272 and _bcExt2 <= 1.414
            if ph(0) > ph(1)
                mScore := -20, mName := "Bearish Cypher", mTarget := pl(0) * 0.786 + ph(0) * 0.214

    // 85 & 86. 5-0 Pattern (CD retraces 50% of BC)
    if mScore == 0

// ============================================================================
var GlobalMatrix GM = GlobalMatrix.new(
     registry              = array.new<PatternManifest>(),
     corr_matrix           = matrix.new<float>(4, 2, 0.0),
     vol_profile_matrix    = matrix.new<float>(20, 2, 0.0),
     htf_volatility_index  = 0.0,
     ltf_precision_factor  = 1.0,
     composite_bias        = 0.0,
     total_patterns_seen   = 0,
     active_pattern_count  = 0
 )

// ============================================================================
// 4. DERIVED PRIMITIVES  (calculated once per bar, reused everywhere)
// ============================================================================
float atr_val   = ta.atr(atr_len)
float vol_mean  = ta.sma(volume, vol_len)
float vol_std   = ta.stdev(volume, vol_len)
float norm_vol  = vol_std != 0 ? (volume - vol_mean) / vol_std : 0.0

// Rolling price-centre (midpoint of highest-high / lowest-low over lookback)
float h_hi      = ta.highest(high,  lookback * 2)
float l_lo      = ta.lowest(low,    lookback * 2)
float price_ctr = math.avg(h_hi, l_lo)

// ============================================================================
// 5. MULTI-TICKER CORRELATION MATRIX  (4 tickers)
// ============================================================================
// request.security calls outside of if-blocks are cached by Pine per-bar
float s1_close = request.security(comp_1, htf_tf, close)
float s2_close = request.security(comp_2, htf_tf, close)
float s3_close = request.security(comp_3, htf_tf, close)
float s4_close = request.security(comp_4, htf_tf, close)

// HTF & LTF derived series
float htf_atr   = request.security(syminfo.tickerid, htf_tf, ta.atr(atr_len))
float ltf_close = request.security(syminfo.tickerid, ltf_tf, close)

// Populate correlation matrix only on the last bar to avoid redundant calls
if barstate.islast
    float c1 = ta.correlation(close, s1_close, corr_len)
    float c2 = ta.correlation(close, s2_close, corr_len)
    float c3 = ta.correlation(close, s3_close, corr_len)
    float c4 = ta.correlation(close, s4_close, corr_len)

    matrix.set(GM.corr_matrix, 0, 0, s1_close) // row=ticker, col=0 → latest close
    matrix.set(GM.corr_matrix, 1, 0, s2_close)
    matrix.set(GM.corr_matrix, 2, 0, s3_close)
    matrix.set(GM.corr_matrix, 3, 0, s4_close)

    matrix.set(GM.corr_matrix, 0, 1, c1)       // col=1 → correlation coefficient
    matrix.set(GM.corr_matrix, 1, 1, c2)
    matrix.set(GM.corr_matrix, 2, 1, c3)
    matrix.set(GM.corr_matrix, 3, 1, c4)

    // Composite bias: average correlation weighted by sign
    float bias_sum = (c1 + c2 + c3 + c4) / 4.0
    GM.composite_bias := bias_sum > 0.3 ? 1.0 : bias_sum < -0.3 ? -1.0 : 0.0

    // HTF volatility index (normalised)
    GM.htf_volatility_index := htf_atr / close * 100.0

    // LTF precision factor (how many LTF bars per current TF bar, clamped)
    float ltf_ratio = close / (ltf_close != 0 ? ltf_close : close)
    GM.ltf_precision_factor := math.max(0.5, math.min(2.0, ltf_ratio))

// ============================================================================
// 6. ROLLING VOLUME-PROFILE MATRIX  (price-bucket → cumulative volume)
//    Updated every bar for an intrabar mini volume-profile
// ============================================================================
// Volume profile updates every bar regardless of registry state
if atr_val > 0
    int   bucket    = int(close / atr_val) % 20  // which of 20 ATR-sized buckets
    float cum_vol   = matrix.get(GM.vol_profile_matrix, bucket, 1) + volume
    matrix.set(GM.vol_profile_matrix, bucket, 0, close)
    matrix.set(GM.vol_profile_matrix, bucket, 1, cum_vol)

// Helper: read POC (point-of-control) price from volume profile
f_poc_price() =>
    float max_vol  = 0.0
    float poc      = close
    for i = 0 to matrix.rows(GM.vol_profile_matrix) - 1
        float bv = matrix.get(GM.vol_profile_matrix, i, 1)
        if bv > max_vol
            max_vol := bv
            poc     := matrix.get(GM.vol_profile_matrix, i, 0)
    poc

// ============================================================================
// 7. POINT & SEGMENT FACTORY FUNCTIONS
// ============================================================================

// Construct a MatrixPoint at a given bar offset
f_make_point(int bi, float price, float v) =>
    float nv  = vol_std != 0 ? (v - vol_mean) / vol_std : 0.0
    MatrixPoint.new(bi, price, v, nv, atr_val)

// Construct a VectorSegment from two MatrixPoints
f_make_segment(MatrixPoint sp, MatrixPoint ep) =>
    float dx      = float(ep.x - sp.x)
    float dy      = ep.y - sp.y
    float slp     = dx != 0 ? dy / dx : 0.0
    float atr_slp = ep.atr_snap != 0 ? slp / ep.atr_snap : 0.0
    float vd      = ep.norm_vol - sp.norm_vol
    bool  exp     = math.abs(ep.y - price_ctr) > math.abs(sp.y - price_ctr)
    bool  valid   = dx > 0 and math.abs(dy) > atr_val * 0.1
    VectorSegment.new(sp, ep, slp, atr_slp, vd, exp, valid)

// Construct a SwingPath from an array of segments
f_make_path(VectorSegment[] segs, int bs, int be) =>
    float total = 0.0
    float slope_sum = 0.0
    for s in segs
        total     += math.abs(s.end_pt.y - s.start_pt.y)
        slope_sum += s.slope
    float avg_slp = array.size(segs) > 0 ? slope_sum / array.size(segs) : 0.0
    bool  imp     = math.abs(avg_slp) > atr_val * 0.05
    SwingPath.new(segs, total, avg_slp, imp, bs, be)

// Weighted confidence score using all 4 correlations + volume quality
f_confidence(float base_corr, float vol_w, bool is_bull) =>
    float c1 = matrix.get(GM.corr_matrix, 0, 1)
    float c2 = matrix.get(GM.corr_matrix, 1, 1)
    float c3 = matrix.get(GM.corr_matrix, 2, 1)
    float c4 = matrix.get(GM.corr_matrix, 3, 1)
    // Mean absolute agreement between tickers
    float avg_corr  = (math.abs(c1) + math.abs(c2) + math.abs(c3) + math.abs(c4)) / 4.0
    // Directional alignment bonus (all tickers agree on direction)
    float dir_sign  = is_bull ? 1.0 : -1.0
    float align     = (c1 * dir_sign + c2 * dir_sign + c3 * dir_sign + c4 * dir_sign) / 4.0
    float raw_score = (avg_corr * 50.0) + (align * 30.0) + (vol_w * 20.0)
    math.max(0.0, math.min(100.0, raw_score))

// 8. PIVOT STACK MANAGEMENT
// ============================================================================
var MatrixPoint[] ph_stack = array.new<MatrixPoint>()
var MatrixPoint[] pl_stack = array.new<MatrixPoint>()

float ph_val = ta.pivothigh(high, lookback, lookback)
float pl_val = ta.pivotlow(low,   lookback, lookback)

if not na(ph_val)
    array.unshift(ph_stack, f_make_point(bar_index - lookback, ph_val, volume[lookback]))
if not na(pl_val)
    array.unshift(pl_stack, f_make_point(bar_index - lookback, pl_val, volume[lookback]))

// Keep stacks trimmed to at most 12 elements (memory cap)
if array.size(ph_stack) > 12
    array.pop(ph_stack)
if array.size(pl_stack) > 12
    array.pop(pl_stack)

// ============================================================================
// 9. FOUR-PATTERN DETECTOR
//    Each detector builds full nested SwingPath → PatternManifest objects.
//    Returns na(PatternManifest) when pattern not found.
// ============================================================================

// --- 9A. MEGA-EXPANSION  (higher-highs + lower-lows divergence) ---
f_detect_expansion(MatrixPoint[] phs, MatrixPoint[] pls) =>
    PatternManifest result = na
    if array.size(phs) >= 3 and array.size(pls) >= 3
        MatrixPoint p0 = array.get(phs, 0)
        MatrixPoint p1 = array.get(phs, 1)
        MatrixPoint p2 = array.get(phs, 2)
        MatrixPoint l0 = array.get(pls, 0)
        MatrixPoint l1 = array.get(pls, 1)
        // Condition: two consecutive HH + two consecutive LL
        if p0.y > p1.y and p1.y > p2.y and l0.y < l1.y
            VectorSegment[] segs_up   = array.new<VectorSegment>()
            VectorSegment[] segs_down = array.new<VectorSegment>()
            segs_up.push(f_make_segment(p2, p1))
            segs_up.push(f_make_segment(p1, p0))
            segs_down.push(f_make_segment(l1, l0))
            SwingPath up_path   = f_make_path(segs_up,   p2.x, p0.x)
            SwingPath down_path = f_make_path(segs_down, l1.x, l0.x)
            bool bullish        = close > l0.y
            float tgt           = p0.y + (p0.y - l0.y) * 1.618
            float stp           = l0.y - atr_val
            float inv           = l1.y - atr_val * 2.0
            float rr            = math.abs(close - stp) != 0 ? math.abs(tgt - close) / math.abs(close - stp) : 0.0
            float vw            = math.max(0.0, math.min(1.0, (p0.norm_vol + l0.norm_vol) / 4.0 + 0.5))
            float conf          = f_confidence(matrix.get(GM.corr_matrix, 0, 1), vw, bullish)
            result := PatternManifest.new("MEGA-EXPANSION", up_path, down_path,
                         tgt, stp, inv, conf, vw, rr, bullish, true,
                         bullish ? color.lime : color.red)
    result

// --- 9B. COMPRESSION  (converging highs and lows → pennant / triangle) ---
f_detect_compression(MatrixPoint[] phs, MatrixPoint[] pls) =>
    PatternManifest result = na
    if array.size(phs) >= 3 and array.size(pls) >= 3
        MatrixPoint p0 = array.get(phs, 0)
        MatrixPoint p1 = array.get(phs, 1)
        MatrixPoint l0 = array.get(pls, 0)
        MatrixPoint l1 = array.get(pls, 1)
        // Condition: lower-highs AND higher-lows (wedge / pennant)
        if p0.y < p1.y and l0.y > l1.y
            VectorSegment[] segs_res = array.new<VectorSegment>()
            VectorSegment[] segs_sup = array.new<VectorSegment>()
            segs_res.push(f_make_segment(p1, p0))
            segs_sup.push(f_make_segment(l1, l0))
            SwingPath res_path = f_make_path(segs_res, p1.x, p0.x)
            SwingPath sup_path = f_make_path(segs_sup, l1.x, l0.x)
            // Breakout direction biased by volume trend
            bool bullish   = l0.norm_vol > p0.norm_vol
            float width    = p0.y - l0.y
            float tgt      = bullish ? p0.y + width * 0.786 : l0.y - width * 0.786
            float stp      = bullish ? l0.y - atr_val       : p0.y + atr_val
            float inv      = bullish ? l1.y - atr_val       : p1.y + atr_val
            float rr       = math.abs(stp - close) != 0 ? math.abs(tgt - close) / math.abs(stp - close) : 0.0
            float vw       = math.max(0.0, math.min(1.0, (l0.norm_vol - p0.norm_vol) * 0.3 + 0.5))
            float conf     = f_confidence(matrix.get(GM.corr_matrix, 1, 1), vw, bullish)
            result := PatternManifest.new("COMPRESSION", res_path, sup_path,
                         tgt, stp, inv, conf, vw, rr, bullish, true,
                         bullish ? color.yellow : color.orange)
    result

// --- 9C. HEAD-AND-SHOULDERS  (three pivots; middle is tallest) ---
f_detect_hs(MatrixPoint[] phs, MatrixPoint[] pls) =>
    PatternManifest result = na
    if array.size(phs) >= 3 and array.size(pls) >= 2
        MatrixPoint ls  = array.get(phs, 2)  // left shoulder
        MatrixPoint hd  = array.get(phs, 1)  // head
        MatrixPoint rs  = array.get(phs, 0)  // right shoulder
        MatrixPoint nl  = array.get(pls, 0)  // neckline anchor
        // Classic H&S: head taller than both shoulders; shoulder symmetry within 1 ATR
        float shoulder_tol = hd.y != 0 ? atr_val / hd.y : 0.05
        bool hs_shape = hd.y > ls.y * 1.005 and hd.y > rs.y * 1.005 and
                        math.abs(ls.y - rs.y) / hd.y < shoulder_tol
        if hs_shape
            VectorSegment[] segs_top = array.new<VectorSegment>()
            VectorSegment[] segs_neck = array.new<VectorSegment>()
            segs_top.push(f_make_segment(ls, hd))
            segs_top.push(f_make_segment(hd, rs))
            segs_neck.push(f_make_segment(nl, f_make_point(rs.x, nl.y, nl.vol)))
            SwingPath top_path  = f_make_path(segs_top,  ls.x, rs.x)
            SwingPath neck_path = f_make_path(segs_neck, nl.x, rs.x)
            float neckline  = nl.y
            float head_ht   = hd.y - neckline
            float tgt       = neckline - head_ht           // measured move
            float stp       = rs.y + atr_val
            float inv       = hd.y + atr_val
            float rr        = math.abs(stp - neckline) != 0 ? math.abs(neckline - tgt) / math.abs(stp - neckline) : 0.0
            float vw        = math.max(0.0, math.min(1.0, (hd.norm_vol - rs.norm_vol) * 0.5 + 0.5))
            float conf      = f_confidence(matrix.get(GM.corr_matrix, 2, 1), vw, false)
            result := PatternManifest.new("HEAD-AND-SHOULDERS", top_path, neck_path,
                         tgt, stp, inv, conf, vw, rr, false, true, color.fuchsia)
    result

// --- 9D. DOUBLE TOP / DOUBLE BOTTOM ---
f_detect_double(MatrixPoint[] phs, MatrixPoint[] pls) =>
    PatternManifest result = na
    // Double Top: two highs within 0.5% of each other
    if array.size(phs) >= 2 and array.size(pls) >= 1
        MatrixPoint t1 = array.get(phs, 1)
        MatrixPoint t2 = array.get(phs, 0)
        MatrixPoint vl = array.get(pls, 0)
        bool dbl_top = t1.y != 0 and math.abs(t1.y - t2.y) / t1.y < 0.005 and t2.x > t1.x
        if dbl_top
            VectorSegment[] segs_t = array.new<VectorSegment>()
            VectorSegment[] segs_v = array.new<VectorSegment>()
            segs_t.push(f_make_segment(t1, t2))
            segs_v.push(f_make_segment(t1, vl))
            SwingPath top_path  = f_make_path(segs_t, t1.x, t2.x)
            SwingPath val_path  = f_make_path(segs_v, t1.x, vl.x)
            float neckline = vl.y
            float height   = t2.y - neckline
            float tgt      = neckline - height
            float stp      = t2.y + atr_val
            float inv      = math.max(t1.y, t2.y) + atr_val * 1.5
            float rr       = height != 0 ? math.abs(neckline - tgt) / (atr_val + height * 0.1) : 0.0
            float vw       = math.max(0.0, math.min(1.0, (vl.norm_vol - t2.norm_vol) * 0.4 + 0.5))
            float conf     = f_confidence(matrix.get(GM.corr_matrix, 3, 1), vw, false)
            result := PatternManifest.new("DOUBLE-TOP", top_path, val_path,
                         tgt, stp, inv, conf, vw, rr, false, true, color.red)

    // Double Bottom: two lows within 0.5%
    if na(result) and array.size(pls) >= 2 and array.size(phs) >= 1
        MatrixPoint b1 = array.get(pls, 1)
        MatrixPoint b2 = array.get(pls, 0)
        MatrixPoint pk = array.get(phs, 0)
        bool dbl_bot = b1.y != 0 and math.abs(b1.y - b2.y) / b1.y < 0.005 and b2.x > b1.x
        if dbl_bot
            VectorSegment[] segs_b = array.new<VectorSegment>()
            VectorSegment[] segs_p = array.new<VectorSegment>()
            segs_b.push(f_make_segment(b1, b2))
            segs_p.push(f_make_segment(b1, pk))
            SwingPath bot_path  = f_make_path(segs_b, b1.x, b2.x)
            SwingPath peak_path = f_make_path(segs_p, b1.x, pk.x)
            float neckline = pk.y
            float height   = neckline - b2.y
            float tgt      = neckline + height
            float stp      = b2.y - atr_val
            float inv      = math.min(b1.y, b2.y) - atr_val * 1.5
            float rr       = height != 0 ? math.abs(tgt - neckline) / (atr_val + height * 0.1) : 0.0
            float vw       = math.max(0.0, math.min(1.0, (b2.norm_vol - pk.norm_vol) * 0.4 + 0.5))
            float conf     = f_confidence(matrix.get(GM.corr_matrix, 0, 1), vw, true)
            result := PatternManifest.new("DOUBLE-BOTTOM", bot_path, peak_path,
                         tgt, stp, inv, conf, vw, rr, true, true, color.aqua)
    result

// ============================================================================
// 10. PATTERN EVALUATION LOOP  (single-pass over all four detectors)
// ============================================================================
// Run detectors
PatternManifest pm_exp  = f_detect_expansion(ph_stack,   pl_stack)
PatternManifest pm_cmp  = f_detect_compression(ph_stack, pl_stack)
PatternManifest pm_hs   = f_detect_hs(ph_stack,          pl_stack)
PatternManifest pm_dbl  = f_detect_double(ph_stack,      pl_stack)

// Collect valid detections in priority order (highest confidence wins per bar)
PatternManifest[] bar_detections = array.new<PatternManifest>()
if not na(pm_exp) and pm_exp.confidence_matrix_score >= min_conf
    bar_detections.push(pm_exp)
if not na(pm_cmp) and pm_cmp.confidence_matrix_score >= min_conf
    bar_detections.push(pm_cmp)
if not na(pm_hs)  and pm_hs.confidence_matrix_score  >= min_conf
    bar_detections.push(pm_hs)
if not na(pm_dbl) and pm_dbl.confidence_matrix_score >= min_conf
    bar_detections.push(pm_dbl)

// Push all new detections into global registry (LIFO, capped at depth)
for pm in bar_detections
    array.unshift(GM.registry, pm)
    GM.total_patterns_seen += 1
    if array.size(GM.registry) > depth
        array.pop(GM.registry)

// Keep active pattern count current (directional stop check per pattern)
int active_cnt = 0
for pm in GM.registry
    bool still_active = pm.is_active and
                        (pm.is_bullish ? close > pm.stop_price : close < pm.stop_price)
    if still_active
        active_cnt += 1
    else
        pm.is_active := false
GM.active_pattern_count := active_cnt

if array.size(bar_detections) > 0

    // --- Polyline through all segment endpoints (no duplicates at junctions) ---
    if show_poly
        chart.point[] draw_pts = array.new<chart.point>()
        for i = 0 to array.size(pm.primary_path.segs) - 1

        // Secondary path polyline (dashed via colour opacity)
        chart.point[] draw_pts2 = array.new<chart.point>()

            VectorSegment seg = array.get(pm.secondary_path.segs, i)
            if i == 0
                draw_pts2.push(chart.point.from_index(seg.start_pt.x, seg.start_pt.y))
            draw_pts2.push(chart.point.from_index(seg.end_pt.x, seg.end_pt.y))
        if array.size(draw_pts2) >= 2
            polyline.new(draw_pts2, curved=false,
                         line_color=color.new(pm.theme_color, 60), line_width=1)

    // --- Target / Stop boxes ---
    if show_boxes
        int box_left  = bar_index - lookback
        int box_right = bar_index + lookback * 3
        // Target box
        box.new(box_left,  pm.target_price + atr_val * 0.25,
                box_right, pm.target_price - atr_val * 0.25,
                bgcolor=color.new(pm.theme_color, 80),
                border_color=pm.theme_color, border_width=1)
        // Stop box
        box.new(box_left,  pm.stop_price + atr_val * 0.25,
                box_right, pm.stop_price - atr_val * 0.25,
                bgcolor=color.new(color.gray, 80),
                border_color=color.gray, border_width=1)
        // Volume-profile POC line (as a thin box)
        float poc_px = f_poc_price()
        box.new(box_left, poc_px + atr_val * 0.1,
                box_right, poc_px - atr_val * 0.1,
                bgcolor=color.new(color.blue, 85),
                border_color=color.new(color.blue, 40), border_width=1)

    // --- Annotation label ---
    if show_labels
        string hud  = "Σ " + pm.signature + "\n" +
                      "CONF: "   + str.tostring(pm.confidence_matrix_score, "#.#") + "%" + "\n" +
                      "R/R: "    + str.tostring(pm.rr_ratio,                "#.##")      + "\n" +
                      "VOL-W: "  + str.tostring(pm.vol_weight,              "#.##")      + "\n" +
                      "TGT: "    + str.tostring(pm.target_price,            format.mintick) + "\n" +
                      "STP: "    + str.tostring(pm.stop_price,              format.mintick) + "\n" +
                      "BIAS: "   + (GM.composite_bias > 0 ? "BULL" : GM.composite_bias < 0 ? "BEAR" : "NEUT")
        label.new(bar_index, high + atr_val, hud,
                  color      = color.new(color.black, 20),
                  textcolor  = pm.theme_color,
                  style      = label.style_label_down,
                  size       = size.small)
