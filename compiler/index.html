Quick Reference

// 1. Configure (before script load — all fields are optional)
window.COMPILER_CONFIG = {
  logToConsole:    true,
  apiEndpoint:     '/compiler/api',   // for remote languages
  defaultLanguage: 'javascript',
  timeout:         10000,
};

// 2. Initialize
MajixCompiler.init();

// 3. Run code
MajixCompiler.run('javascript', 'console.log("hello")')
  .then(function (result) {
    console.log(result.stdout);   // "hello"
    console.log(result.elapsed);  // ms
  });

// 4. Compile / syntax-check
MajixCompiler.compile('javascript', 'const x = ;')
  .then(function (result) {
    if (result.status === 'error') console.error(result.stderr);
  });

// 5. Remote language (requires API endpoint)
MajixCompiler.run('python', 'print("hello from Python")')
  .then(function (r) { console.log(r.stdout); });

// 6. Register a custom adapter
MajixCompiler.register('brainfuck', {
  canRun: true,
  run: function (code, options) {
    return Promise.resolve({ language:'brainfuck', status:'ok',
      stdout: '(brainfuck output)', stderr:'', exitCode:0,
      elapsed:0, output:null, error:null });
  }
}, ['bf']);

// 7. Inspect
MajixCompiler.getSupportedLanguages();   // ['javascript', 'html', 'json', 'python', ...]
MajixCompiler.isSupported('rust');       // true
MajixCompiler.history(5);               // last 5 results (most recent first)
MajixCompiler.clearHistory();
